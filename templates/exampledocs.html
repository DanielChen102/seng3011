Example Usage

Below are examples of using the API in other systems

Using JQuery from JavaScript 

    var parameters = {
    'company':'WOW.AX',
    'start': '2015-10-01T08:45:10.295Z',
    'end': '2015-11-01T19:37:12.193Z'
    'fields': [
        'id',
        'name',
        'website',
        'fan_count'
        ]
    };

    // encode parameters in request url
    var url = 'http://seng3011laser.com/api/v1/PageData?' + jQuery.param(parameters);

    jQuery.ajax({ url: url, dataType: 'json', contentType: "application/json", success: function(data){ console.log(data); } });

Using Fetch from JavaScript 

    var url = 'http://seng3011laser.com/v1/company/id/WOW.AX?start=2015-10-01T08:45:10.295Z&end=2015-11-01T19:37:12.193Z&fields=id,name,website,fan_count,post_like_count';

    fetch(url, {
        method: 'Get',
        Headers: {
            'Accept': 'application/json',
            'Content-Type': 'application/json'
        }
    })
    .then(function(response) {
        return response.json();
    })

Python 

    import requests
    from urllib.parse import urlencode

    parameters = {'company':'WOW.AX', 'start': '2015-10-01T08:45:10.295Z', 'end': '2015-11-01T19:37:12.193Z', 'fields': ['id', 'name', 'website', 'fan_count']}

    # encode parameters in request url
    url = 'http://seng3011laser.com/api/v1/PageData?' + urlencode(parameters)

    # send request, get response
    response = requests.get(url)

    # print response
    print(json.loads(response))

Java 

    import java.io.BufferedReader;
    import java.io.InputStreamReader;
    import java.net.HttpURLConnection;
    import java.net.URL;

	// HTTP GET request
	private void sendGet() throws Exception {

		String url = "http://seng3011laser.com/v1/company/id/WOW.AX?start=
        2015-10-01T08:45:10.295Z&end=2015-11-01T19:37:12.193Z&fields=id,
        name,website,fan_count,post_like_count";

		URL obj = new URL(url);
		HttpURLConnection con = (HttpURLConnection) obj.openConnection();

		// optional default is GET
		con.setRequestMethod("GET");

		//add request header
		con.setRequestProperty("User-Agent", USER_AGENT);

		int responseCode = con.getResponseCode();
		System.out.println("\nSending 'GET' request to URL : " + url);
		System.out.println("Response Code : " + responseCode);

		BufferedReader in = new BufferedReader(
		        new InputStreamReader(con.getInputStream()));
		String inputLine;
		StringBuffer response = new StringBuffer();

		while ((inputLine = in.readLine()) != null) {
			response.append(inputLine);
		}
		in.close();

		//print result
		System.out.println(response.toString());

	}
